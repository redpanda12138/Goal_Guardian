# 方案二实施计划：MAS作为独立服务，talkieai-server作为网关

## 一、架构设计

### 1.1 整体架构

```
┌─────────────────────────────────────────────────────────┐
│              talkieai-server (网关层)                    │
│  - 端口: 8097 (默认)                                     │
│  - 统一认证 (JWT)                                        │
│  - API路由转发                                           │
│  - account_id ↔ patient_id 映射                         │
└─────────────────────────────────────────────────────────┘
                        │
        ┌───────────────┼───────────────┐
        │               │               │
        ▼               ▼               ▼
┌──────────┐    ┌──────────┐    ┌──────────┐
│   MMA    │    │   SOA     │    │   GRA    │
│ 端口8001 │    │ 端口8002  │    │ 端口8003 │
└──────────┘    └──────────┘    └──────────┘
        │               │               │
        └───────────────┼───────────────┘
                        │
        ┌───────────────┼───────────────┐
        │               │               │
        ▼               ▼               ▼
┌──────────┐    ┌──────────┐    ┌──────────┐
│   SCA    │    │   SSA     │    │   OA     │
│ 端口8004 │    │ 端口8005  │    │ 端口8006 │
└──────────┘    └──────────┘    └──────────┘
```

### 1.2 数据流

```
客户端请求
    ↓
talkieai-server (认证 + account_id → patient_id映射)
    ↓
转发到MAS服务 (使用patient_id)
    ↓
MAS服务处理 (使用JSON文件存储)
    ↓
返回响应
    ↓
talkieai-server (可选：数据同步到数据库)
    ↓
返回给客户端
```

#### 1.2 实现映射服务

**文件**: `app/services/mas/patient_mapping_service.py`

```python
from sqlalchemy.orm import Session
from app.db.mas_entities import PatientMappingEntity
from app.core.utils import short_uuid

class PatientMappingService:
    def __init__(self, db: Session):
        self.db = db
    
    def get_or_create_patient_id(self, account_id: str) -> str:
        """获取或创建patient_id"""
        mapping = self.db.query(PatientMappingEntity).filter_by(
            account_id=account_id
        ).first()
        
        if mapping:
            return mapping.patient_id
        
        # 创建新的映射
        patient_id = f"patient_{short_uuid()}"
        mapping = PatientMappingEntity(
            id=f"mapping_{short_uuid()}",
            account_id=account_id,
            patient_id=patient_id
        )
        self.db.add(mapping)
        self.db.commit()
        return patient_id
    
    def get_account_id(self, patient_id: str) -> str:
        """根据patient_id获取account_id"""
        mapping = self.db.query(PatientMappingEntity).filter_by(
            patient_id=patient_id
        ).first()
        return mapping.account_id if mapping else None
```

### 阶段二：网关服务层（1-2周）

#### 2.1 创建MAS网关服务

**文件**: `app/services/mas/mas_gateway_service.py`

```python
import httpx
from typing import Dict, Any, Optional
from app.config import Config
from app.core.logging import logging

class MASGatewayService:
    """MAS服务网关，负责转发请求到各个MAS服务"""
    
    # MAS服务地址配置（可通过环境变量配置）
    MAS_SERVICES = {
        "mma": Config.MAS_MMA_URL or "http://localhost:8001",
        "soa": Config.MAS_SOA_URL or "http://localhost:8002",
        "gra": Config.MAS_GRA_URL or "http://localhost:8003",
        "sca": Config.MAS_SCA_URL or "http://localhost:8004",
        "ssa": Config.MAS_SSA_URL or "http://localhost:8005",
        "oa": Config.MAS_OA_URL or "http://localhost:8006"
    }
    
    @staticmethod
    async def call_mas_service(
        service_name: str,
        endpoint: str,
        method: str = "POST",
        data: Optional[Dict[str, Any]] = None,
        timeout: int = 30
    ) -> Dict[str, Any]:
        """调用MAS服务"""
        base_url = MASGatewayService.MAS_SERVICES.get(service_name.lower())
        if not base_url:
            raise ValueError(f"Unknown MAS service: {service_name}")
        
        url = f"{base_url}{endpoint}"
        
        try:
            async with httpx.AsyncClient(timeout=timeout) as client:
                if method.upper() == "GET":
                    response = await client.get(url, params=data)
                else:
                    response = await client.post(url, json=data)
                
                response.raise_for_status()
                return response.json()
        except httpx.HTTPError as e:
            logging.error(f"MAS service call failed: {service_name} {endpoint} - {e}")
            raise Exception(f"MAS服务调用失败: {str(e)}")
    
    @staticmethod
    async def check_service_health(service_name: str) -> bool:
        """检查MAS服务健康状态"""
        try:
            base_url = MASGatewayService.MAS_SERVICES.get(service_name.lower())
            async with httpx.AsyncClient(timeout=5) as client:
                response = await client.get(f"{base_url}/health", timeout=5)
                return response.status_code == 200
        except:
            return False
```

#### 3.1 创建MAS路由

**文件**: `app/api/mas_routes.py`

```python
from fastapi import APIRouter, Depends, Request
from sqlalchemy.orm import Session
from typing import Optional
from app.core import get_current_account
from app.db import get_db
from app.models.response import ApiResponse
from app.services.mas.patient_mapping_service import PatientMappingService
from app.services.mas.mas_gateway_service import MASGatewayService
from pydantic import BaseModel

router = APIRouter()

# ========== 数据模型 ==========
class SubmitNotesDTO(BaseModel):
    """提交健康教练会话笔记"""
    notes: list  # 会话笔记列表

class SendMessageDTO(BaseModel):
    """发送消息到当前代理"""
    user_input: str
    turn_index: int

class TriggerSessionDTO(BaseModel):
    """触发会话"""
    patient_id: Optional[str] = None  # 可选，默认使用当前用户

# ========== 患者信息管理 ==========
@router.post("/mas/patients/notes", name="Submit Session Notes")
async def submit_notes(
    dto: SubmitNotesDTO,
    db: Session = Depends(get_db),
    account_id: str = Depends(get_current_account)
):
    """提交健康教练会话笔记，触发MMA提取"""
    mapping_service = PatientMappingService(db)
    patient_id = mapping_service.get_or_create_patient_id(account_id)
    
    # 转换数据格式：添加patient_id到每个笔记
    notes_data = []
    for note in dto.notes:
        note_data = note.copy()
        note_data["study_id"] = patient_id
        notes_data.append(note_data)
    
    # 调用MMA服务
    result = await MASGatewayService.call_mas_service(
        "mma",
        "/extract",
        data=notes_data
    )
    
    return ApiResponse(data=result)

@router.get("/mas/patients/info", name="Get Patient Info")
async def get_patient_info(
    db: Session = Depends(get_db),
    account_id: str = Depends(get_current_account)
):
    """获取患者结构化信息"""
    mapping_service = PatientMappingService(db)
    patient_id = mapping_service.get_or_create_patient_id(account_id)
    
    result = await MASGatewayService.call_mas_service(
        "mma",
        f"/patient_notes/{patient_id}",
        method="GET"
    )
    
    return ApiResponse(data=result)

@router.get("/mas/patients/goals", name="Get Patient Goals")
async def get_patient_goals(
    db: Session = Depends(get_db),
    account_id: str = Depends(get_current_account)
):
    """获取患者SMART目标"""
    mapping_service = PatientMappingService(db)
    patient_id = mapping_service.get_or_create_patient_id(account_id)
    
    result = await MASGatewayService.call_mas_service(
        "mma",
        f"/patient_goals/{patient_id}",
        method="GET"
    )
    
    return ApiResponse(data=result)

# ========== 会话管理 ==========
@router.post("/mas/sessions/trigger", name="Trigger Session")
async def trigger_session(
    dto: TriggerSessionDTO,
    db: Session = Depends(get_db),
    account_id: str = Depends(get_current_account)
):
    """手动触发健康教练会话（SOA）"""
    mapping_service = PatientMappingService(db)
    patient_id = dto.patient_id or mapping_service.get_or_create_patient_id(account_id)
    
    result = await MASGatewayService.call_mas_service(
        "soa",
        "/trigger",
        data={"patient_id": patient_id, "turn_index": 1}
    )
    
    return ApiResponse(data=result)

@router.post("/mas/sessions/message", name="Send Message")
async def send_message(
    dto: SendMessageDTO,
    db: Session = Depends(get_db),
    account_id: str = Depends(get_current_account)
):
    """发送消息到当前代理"""
    mapping_service = PatientMappingService(db)
    patient_id = mapping_service.get_or_create_patient_id(account_id)
    
    # 确定当前是哪个代理（需要从OA获取会话状态）
    # 这里简化处理，实际需要查询OA的会话状态
    
    # 先尝试SOA
    try:
        result = await MASGatewayService.call_mas_service(
            "soa",
            "/receive_message",
            data={
                "patient_id": patient_id,
                "user_input": dto.user_input,
                "turn_index": dto.turn_index
            }
        )
        return ApiResponse(data=result)
    except:
        # 如果SOA失败，尝试GRA
        try:
            result = await MASGatewayService.call_mas_service(
                "gra",
                "/receive_message",
                data={
                    "patient_id": patient_id,
                    "user_input": dto.user_input,
                    "turn_index": dto.turn_index
                }
            )
            return ApiResponse(data=result)
        except:
            # 最后尝试SCA
            result = await MASGatewayService.call_mas_service(
                "sca",
                "/receive_message",
                data={
                    "patient_id": patient_id,
                    "user_input": dto.user_input,
                    "turn_index": dto.turn_index
                }
            )
            return ApiResponse(data=result)

@router.get("/mas/sessions/current", name="Get Current Session")
async def get_current_session(
    db: Session = Depends(get_db),
    account_id: str = Depends(get_current_account)
):
    """获取当前会话状态（从OA获取）"""
    mapping_service = PatientMappingService(db)
    patient_id = mapping_service.get_or_create_patient_id(account_id)
    
    # 从OA获取会话状态（需要OA提供此接口）
    # 这里需要根据实际OA的接口调整
    result = await MASGatewayService.call_mas_service(
        "oa",
        f"/session_status/{patient_id}",
        method="GET"
    )
    
    return ApiResponse(data=result)

# ========== 预约管理 ==========
@router.post("/mas/schedules", name="Create Schedule")
async def create_schedule(
    dto: SubmitNotesDTO,  # 复用，实际应该是预约数据
    db: Session = Depends(get_db),
    account_id: str = Depends(get_current_account)
):
    """创建/更新预约（通知OA）"""
    mapping_service = PatientMappingService(db)
    patient_id = mapping_service.get_or_create_patient_id(account_id)
    
    # 转换数据格式
    schedule_data = []
    for note in dto.notes:
        schedule_data.append({
            "study_id": patient_id,
            "date": note.get("date")
        })
    
    result = await MASGatewayService.call_mas_service(
        "oa",
        "/new_sessions",
        data=schedule_data
    )
    
    return ApiResponse(data=result)
```
